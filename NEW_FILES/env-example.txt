# ElectroDante Environment Variables
# Copy this file to .env and fill in your values

# ===================================
# SUPABASE CONFIGURATION (REQUIRED)
# ===================================
# Get these from your Supabase project settings
VITE_SUPABASE_URL=https://your-project.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-key-here

# Service role key (server-side only, NOT exposed to browser)
VITE_SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# ===================================
# ENCRYPTION KEY (REQUIRED)
# ===================================
# Generate a random 32-character key for encrypting local storage
# You can generate one with: openssl rand -hex 16
VITE_ENCRYPTION_KEY=your-32-char-encryption-key-here

# ===================================
# DANTE BRIDGE SERVER (REQUIRED)
# ===================================
# WebSocket URL for your Railway deployment
# Use wss:// for production (HTTPS), ws:// for local development
